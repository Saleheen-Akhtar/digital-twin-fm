# apps/web (Next.js 15 Dashboard) Scaffolding Plan

> **For Hermes:** Use subagent-driven-development skill to implement this plan task-by-task. **LOCAL ONLY — do not push to remote without explicit user approval.**

**Goal:** Scaffold a runnable Next.js 15 dashboard at `apps/web` that calls the existing `apps/api-gateway` `/health` and `/auth/login` endpoints, with Infisical-aware env loading, Tailwind CSS, a login page, and a basic dashboard shell.

**Architecture:** Next.js 15 with the App Router and React Server Components. The app loads secrets via a `@digital-twin-fm/web/env` module that mirrors the api-gateway Infisical pattern. Server data fetching uses native `fetch` with revalidation (no React Query yet — that's added when a domain needs it). UI styling uses Tailwind CSS 4 (Vite-style, no config file needed). The first two routes are `/login` (uses the api-gateway `/auth/login`) and `/dashboard` (a placeholder shell that shows the connected user's name and a health check from the api-gateway).

**Tech Stack:** Next.js 15, React 19, TypeScript 5, Tailwind CSS 4, `@digital-twin-fm/db` (workspace), `@digital-twin-fm/web/env` (local module), `pnpm`, `turbo`.

/**
 * Web app environment configuration.
 *
 * - `getServerEnv()`: only available in Server Components, Route Handlers,
 *   and Server Actions. Reads `API_GATEWAY_URL` (server-to-server).
 * - `getClientEnv()`: available everywhere. Reads `NEXT_PUBLIC_*` vars
 *   that Next.js inlines into the client bundle at build time.
 *
 * For secrets (Infisical) the api-gateway already handles them; this web
 * app only needs public URLs.
 */

function requireEnv(name: string, devFallback: string): string {
  const value = process.env[name];
  if (value && value.length > 0) return value;
  if (process.env.NODE_ENV !== 'production') return devFallback;
  throw new Error(`Required env var ${name} is missing in production.`);
}

export function getServerEnv() {
  return {
    apiGatewayUrl: requireEnv('API_GATEWAY_URL', 'http://localhost:4000'),
  };
}

export function getClientEnv() {
  return {
    apiBaseUrl: requireEnv('NEXT_PUBLIC_API_URL', 'http://localhost:4000'),
  };
}
```

/**
 * Thin fetch wrapper for the api-gateway. Server-side only.
 *
 * Injects the base URL from `getServerEnv()`, sends JSON, and surfaces
 * non-2xx responses as thrown Errors with the server's error message.
 */

export interface ApiClientOptions {
  baseUrl: string;
}

export interface ApiClientDeps {
  /** Injectable for tests. Defaults to global `fetch`. */
  fetch?: typeof fetch;
}

export interface LoginInput {
  email: string;
  password: string;
}

export function createApiClient(opts: ApiClientOptions, deps: ApiClientDeps = {}) {
  const f = deps.fetch ?? fetch;
  const base = opts.baseUrl.replace(/\/$/, '');

  async function call<T>(path: string, init: RequestInit = {}): Promise<T> {
    const res = await f(`${base}${path}`, {
      ...init,
      headers: {
        'content-type': 'application/json',
        ...(init.headers ?? {}),
      },
    });
    if (!res.ok) {
      let msg = `HTTP ${res.status}`;
      try {
        const body = await res.json();
        if (body?.message) msg = String(body.message);
      } catch {
        // ignore
      }
      throw new Error(msg);
    }
    return (await res.json()) as T;
  }

  return {
    health: () => call<{ status: string }>('/health', { method: 'GET' }),
    login: (input: LoginInput) =>
      call<{ accessToken: string }>('/auth/login', {
        method: 'POST',
        body: JSON.stringify(input),
      }),
  };
}
```

'use client';

import { useState, type FormEvent } from 'react';

export function LoginForm({ error }: { error: string | null }) {
  const [pending, setPending] = useState(false);

  async function onSubmit(e: FormEvent<HTMLFormElement>) {
    e.preventDefault();
    setPending(true);
    const fd = new FormData(e.currentTarget);
    // POST to the server action endpoint via standard form submission.
    // The server action handles the api-gateway call.
    const body = new URLSearchParams();
    body.set('email', String(fd.get('email') ?? ''));
    body.set('password', String(fd.get('password') ?? ''));
    const res = await fetch('/login', { method: 'POST', body });
    if (res.redirected) {
      window.location.href = res.url;
      return;
    }
    setPending(false);
  }

  return (
    <form onSubmit={onSubmit} className="flex flex-col gap-4 max-w-sm w-full">
      {error && (
        <div role="alert" className="text-red-400 text-sm">
          {error}
        </div>
      )}
      <label className="flex flex-col gap-1">
        <span className="text-sm">Email</span>
        <input
          name="email"
          type="email"
          required
          autoComplete="username"
          className="border border-neutral-700 rounded px-3 py-2 bg-neutral-900"

Write `apps/web/src/app/login/actions.ts`:
```ts
'use server';

import { cookies } from 'next/headers';
import { redirect } from 'next/navigation';
import { getServerEnv } from '@/env';
import { createApiClient } from '@/lib/api-client';

export interface LoginState {
  error: string | null;
}

export async function loginAction(_prev: LoginState, formData: FormData): Promise<LoginState> {
  const email = String(formData.get('email') ?? '');
  const password = String(formData.get('password') ?? '');

  if (!email || !password) {
    return { error: 'Email and password are required.' };
  }

  const { apiGatewayUrl } = getServerEnv();
  const api = createApiClient({ baseUrl: apiGatewayUrl });

  try {
    const { accessToken } = await api.login({ email, password });
    (await cookies()).set('dtfm_token', accessToken, {
      httpOnly: true,
      secure: process.env.NODE_ENV === 'production',
      sameSite: 'lax',
      path: '/',
      maxAge: 60 * 60 * 8, // 8 hours
    });
  } catch (err) {
    const message = err instanceof Error ? err.message : 'Login failed';
    return { error: message };
  }

  redirect('/dashboard');
}
```
